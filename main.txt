import numpy as np
def nott(a):
    return 1-a#-|
def and_(a,b):
    return np.fmin(a,b)#\/
def orr(a,b):
    return np.fmax(a,b)#^
def impc(a,b):
    return np.fmax(np.fmin(a,b),(1-a))#->
def eq(a,b):
    return np.fmin(np.fmax(nott(a),b),np.fmax(a,nott(b)))
a,b,c=0.3,0.7,0.5
print(eq(impc(and_(a,b),c),and_(impc(a,c),nott(c))))#3
